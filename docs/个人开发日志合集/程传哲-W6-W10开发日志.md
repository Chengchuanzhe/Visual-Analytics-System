```javascript
window.addEventListener('resize', debounce(function() {
  const width = window.innerWidth;
  
  if (width > 1200) {
    // 大屏: 2列布局
    document.querySelector('.chart-grid').style.gridTemplateColumns = '1fr 1fr';
  } else if (width > 768) {
    // 中屏: 1列布局
    document.querySelector('.chart-grid').style.gridTemplateColumns = '1fr';
  } else {
    // 小屏: 1列布局, 调整间距
    document.querySelector('.chart-grid').style.gap = '12px';
  }
}, 300));

// 防抖函数
function debounce(func, delay) {
  let timeout;
  return function(...args) {
    clearTimeout(timeout);
    timeout = setTimeout(() => func(...args), delay);
  };
}
```

```javascript
// 系统初始化
document.addEventListener('DOMContentLoaded', function() {
  // 1. 加载颜色方案
  const colorScheme = getUnifiedColorScheme();
  
  // 2. 初始化导航
  initNavigation();
  
  // 3. 初始化侧边栏
  initSidebar();
  
  // 4. 加载数据统计
  loadDataStats();
});

// 导航初始化
function initNavigation() {
  const tabButtons = document.querySelectorAll('.tab-btn');
  
  tabButtons.forEach((btn, index) => {
    if (index === 0) {
      btn.classList.add('active');
    }
    
    btn.addEventListener('click', function(e) {
      e.preventDefault();
      
      // 更新导航按钮状态
      tabButtons.forEach(b => b.classList.remove('active'));
      this.classList.add('active');
      
      // 显示对应图表
      const member = this.dataset.member;
      switchMember(member);
    });
  });
}

// 成员切换
function switchMember(member) {
  // 隐藏所有容器
  const containers = document.querySelectorAll('.chart-container');
  containers.forEach(container => {
    container.classList.add('hidden');
  });
  
  // 显示选中容器
  const activeContainer = document.getElementById(`member-${member}`);
  activeContainer.classList.remove('hidden');
  
  // 更新侧边栏
  updateSidebar(member);
}

// 侧边栏初始化
function initSidebar() {
  const legend = document.getElementById('legend');
  const colorScheme = getUnifiedColorScheme();
  
  colorScheme.regions.forEach(region => {
    const item = document.createElement('div');
    item.className = 'legend-item';
    
    const colorBox = document.createElement('div');
    colorBox.className = 'legend-color';
    colorBox.style.backgroundColor = region.hex;
    
    const label = document.createElement('span');
    label.textContent = region.name;
    label.className = 'legend-label';
    
    item.appendChild(colorBox);
    item.appendChild(label);
    
    legend.appendChild(item);
  });
}

// 数据统计加载
function loadDataStats() {
  // 从JSON获取数据统计
  fetch('data/dataset.json')
    .then(response => response.json())
    .then(data => {
      // 更新显示
      document.getElementById('total-records').textContent = data.length;
      
      // 计算地域数和维度数
      const regions = new Set(data.map(d => d.region)).size;
      const dimensions = Object.keys(data[0]).length - 1; // 除去id
      
      document.getElementById('regions-count').textContent = regions;
      document.getElementById('dimensions-count').textContent = dimensions;
    });
}

// 获取统一颜色方案
function getUnifiedColorScheme() {
  return {
    regions: [
      {name: '太平洋', hex: '#0173B2'},
      {name: '大西洋', hex: '#DE8F05'},
      {name: '印度洋', hex: '#029E73'},
      {name: '北极', hex: '#ECE2F0'},
      {name: '南极', hex: '#56B4E9'},
      {name: '赤道', hex: '#A6761D'}
    ]
  };
}
```
